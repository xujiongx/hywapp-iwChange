>发送图片到后端
<!-- 

* @data json     发送的数据
* @return json   返回的数据

-->

``` js
data: {
    message: '发送主播选定图片地址',
    imgArr: {
        imageP,
        imageK
    }
}

return :{
    message: 'success', //返回的消息
    code: 200, //200是成功
    data: {
        //没有数据就是空，有数据就是数据
    }
}
```

``` js
  setImageArr = () => {
      let {
          imageP,
          imageK
      } = this.state
      let args = {
          url: '',
          method: 'POST',
          data: {
              message: '发送主播选定图片地址',
              imgArr: {
                  imageP,
                  imageK
              }
          },
          dataType: 'json'
      }
      hyExt.request(args)
  }
```

> 向后端请求图片地址：

<!-- 

* @data json  
* @return json

-->

``` js
data: {
    message: '请求用户选定图片',
}

return :{
    message: '',
    code: '',
    data: {
        imageP: '',
        imageK: ''
    }
}
```

``` js
getShowImages = () => {
    let obj = {
        url: '',
        method: 'GET',
        dataType: 'json'
    }
    hyExt.request(obj).then(res => {
        this.setState({
            showImageP: res.data.imageP,
            showImageK: res.data.imageK,
        })
    }).catch(err => {
        console.log(err);
    })
}
```

> 用户点赞

<!-- 

* @data json  
* @return json

-->

``` js
data: {
    message: '用户点赞',
    userId:null,
    userInfo,
    imageSelect: 0 / 1, //等于0是当前主播当前图片数组列表第一张，等与1是第二张
    number: 1, //点赞增加的票数
    countNumber:null
}

userInfo: { //用户详细信息，包括头像，昵称,等级
    userNick: '',
    userAvatarUrl: '',
    userLevel: ''
}

return :{
    message: '',
    code: '',
    data: {

    }
}
```

``` js
let {
    colorCheck,
    userInfo,
} = this.state
let obj = {
    url: '',
    method: 'POST',
    data: {
        message: '用户点赞'，
        userId:null,
        userInfo,
        imageSelect: colorCheck
        number: 1,
        countNumber:null
    }
    dataType: 'json'
}
hyExt.request(obj)
```

return :{
